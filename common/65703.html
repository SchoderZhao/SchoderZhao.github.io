1:接口文档上传配置

{
"inlineEnum":true,//设置为true会将枚举详情展示到参数表中，默认关闭，//@since 1.8.8版本开始
"displayActualType":false,//配置true会在注释栏自动显示泛型的真实类型短类名，@since 1.9.6
"outPath": "D://md2", //指定文档的输出路径
"appKey": "",// torna空间开放用户的appKey 管理员生成
"appToken": "", //torna空间项目模块的token
"secret": "",//torna空间开放用户的secret 管理员生成 
"openUrl": "http://192.168.90.163:7700/api",//torna平台地址(目前为固定值)
"debugEnvName":"测试环境", //torna环境名称
"debugEnvUrl":"http://192.168.90.208:8080",//torna接口测试的地址
"errorCodeDictionaries": [{ //错误码列表，没有需求可以不设置
"enumClassName": "com.itjcloud.common.result.ResultEnum", //错误码枚举类
"codeField": "code",//错误码的code码字段名称
"descField": "msg"//错误码的描述信息对应的字段名
}]
}
